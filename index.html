<script>

(function(){

  const enableAutoplay = false;     // set to true if you want auto-advance

  const autoplayMs     = 5200;


  const pills = [...document.querySelectorAll('.pill')];

  const cards = [...document.querySelectorAll('.card')];

  const prev  = document.getElementById('prev');

  const next  = document.getElementById('next');


  let i = 0, lock = false, timer = null;


  function setPills(idx){

    pills.forEach((p,j)=>{

      p.classList.toggle('active', j===idx);

      p.classList.toggle('side',   j!==idx);

      p.setAttribute('aria-selected', j===idx ? 'true' : 'false');

    });

  }


  function show(idx){

    if (lock) return;

    idx = (idx + cards.length) % cards.length;

    if (idx === i) return;

    lock = true;


    cards[i].classList.remove('show');

    cards[idx].classList.add('show');

    i = idx; setPills(i);

    resizeToFit();


    setTimeout(()=>{ lock=false; }, 380);

  }


  function go(n){ show(n); if (timer) { clearInterval(timer); startAutoplay(); } }


  pills.forEach(p => p.addEventListener('click', () => go(Number(p.dataset.idx)), {passive:true}));

  prev.addEventListener('click', () => go(i-1), {passive:true});

  next.addEventListener('click', () => go(i+1), {passive:true});

  document.addEventListener('keydown', e => {

    if (e.key === 'ArrowLeft')  go(i-1);

    if (e.key === 'ArrowRight') go(i+1);

  });


  function startAutoplay(){ if(!enableAutoplay) return; timer=setInterval(()=>show(i+1), autoplayMs); }

  function stopAutoplay(){ if(timer) clearInterval(timer); timer=null; }


  /* Post height to parent so iframe auto-resizes */

  function resizeToFit(){

    const h = document.documentElement.scrollHeight;

    parent?.postMessage?.({type:'ihPillarsHeight', height:h}, '*');

  }

  new ResizeObserver(resizeToFit).observe(document.body);

  window.addEventListener('load', resizeToFit);

  window.addEventListener('resize', resizeToFit);


  // init

  requestAnimationFrame(()=>{ cards.forEach((c,idx)=>c.classList.toggle('show', idx===0)); setPills(0); resizeToFit(); startAutoplay(); });

})();

</script>
